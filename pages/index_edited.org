     (-filter (-lambda ((&hash :draft-p :edited-date :type))
		  (and edited-date
		   (not draft-p)
		   (-contains-p '("post" "note") type))))

	      (ht-get it :edited-date)
	      (ht-get other :edited-date)))

	       (format "- <%s> [[file:./%s.html][%s]]"
		edited-date (f-base html-dest) title)))
     (s-join "\n"))
